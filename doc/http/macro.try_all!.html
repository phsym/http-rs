macro_rules! try_all {
	(
		$($expr:expr); *;
	) => (
		{
			$(
				try!($expr);
			)*
		}
	)
}

Wrap all the given expressions in a try!

Example

fn do_something() -> Result<String, Error> {
    return Ok("Ok".to_string());
}

try_all! {
    do_something();
    do_something();
    do_something();
};
